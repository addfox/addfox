# Preact

Add `@rsbuild/plugin-preact` to **plugins** in `addfox.config.ts` to enable Preact and JSX/TSX.

## Setup

<Tabs>
  <Tab label={<TabLabelPnpm />}>

```bash
pnpm add preact
pnpm add @rsbuild/plugin-preact -D
```

  </Tab>
  <Tab label={<TabLabelNpm />}>

```bash
npm install preact
npm install @rsbuild/plugin-preact -D
```

  </Tab>
  <Tab label={<TabLabelYarn />}>

```bash
yarn add preact
yarn add @rsbuild/plugin-preact -D
```

  </Tab>
  <Tab label={<TabLabelBun />}>

```bash
bun add preact
bun add @rsbuild/plugin-preact -D
```

  </Tab>
</Tabs>

```ts
// addfox.config.ts
import { defineConfig } from "addfox";
import { pluginPreact } from "@rsbuild/plugin-preact";

export default defineConfig({
  plugins: [pluginPreact()],
});
```
